15.SELECT YEAR, COUNT(SOLD)
   FROM BOOK
   GROUP BY YEAR
16.SELECT GENRE, AVG(SOLD), DESCRIPTION
   FROM BOOK, GENRE
   WHERE GENRE=GENRE_ID
   GROUP BY GENRE
17.SELECT AUTHOR, SUM(SOLD), FIRST_NAME, LAST_NAME
   FROM BOOK, AUTHOR
   WHERE AUTHOR=AUTHOR_ID
   GROUP BY AUTHOR
18.SELECT AUTHOR, LAST_NAME, FIRST_NAME, COUNT(*) 
   FROM BOOK, AUTHOR
   WHERE AUTHOR=AUTHOR_ID
   GROUP BY AUTHOR
   HAVING COUNT(*) >=1
19.SELECT COUNT(*), NAME
   FROM BOOK, PUBLISHER 
   WHERE PUBLISHER=PUBL_ID
   GROUP BY PUBLISHER
   HAVING COUNT(*) >=3
