16.
SELECT DESCRIPTION, AVG(SOLD)
FROM BOOK
LEFT JOIN GENRE
ON GENRE = GENRE_ID
GROUP BY DESCRIPTION

17.
SELECT FIRST_NAME, LAST_NAME, SUM(SOLD) 
FROM AUTHOR
LEFT JOIN BOOK
ON AUTHOR = AUTHOR_ID
GROUP BY AUTHOR

18. 
SELECT LAST_NAME , FIRST_NAME , COUNT(*)
FROM AUTHOR
LEFT JOIN BOOK
ON AUTHOR = AUTHOR_ID
GROUP BY AUTHOR
HAVING COUNT(*) > 1

ON PUBLISHER = PUBL_ID
GROUP BY PUBLISHER
