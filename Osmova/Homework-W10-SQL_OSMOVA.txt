Q16
SELECT DESCRIPTION, AVG(SOLD)
FROM BOOK
LEFT JOIN GENRE
ON GENRE = GENRE_ID
GROUP BY GENRE;

Q17
SELECT FIRST_NAME,  LAST_NAME, SUM(SOLD)
FROM BOOK
LEFT JOIN AUTHOR
ON AUTHOR = AUTHOR_ID
GROUP BY AUTHOR;

Q18
SELECT AUTHOR, LAST_NAME, FIRST_NAME, COUNT(*)
FROM AUTHOR
LEFT JOIN BOOK
ON AUTHOR = AUTHOR_ID
GROUP BY AUTHOR
HAVING COUNT(*) > 1;
